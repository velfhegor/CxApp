const express = require('express'); const app = express(); app.get('/stock/:ticker', async (req, res) => { // 1. Grab whatever ticker symbol tracker2.html is searching for const ticker = req.params.ticker.toUpperCase(); try { // 2. Pass that exact ticker directly to Yahoo's public data engine const targetUrl = `https://query1.finance.yahoo.com/v8/finance/chart/${ticker}?interval=1d&range=2d`; const response = await fetch(targetUrl, { headers: { 'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64)' } }); if (!response.ok) throw new Error(`Data source returned status ${response.status}`); const data = await response.json(); // 3. Make sure the data exists, then return it in the exact layout your script expects if (data && data.chart && data.chart.result && data.chart.result[0]) { const meta = data.chart.result[0].meta; const fakeYahooStructure = { chart: { result: [ { meta: { regularMarketPrice: meta.regularMarketPrice || meta.chartPreviousClose, previousClose: meta.previousClose || meta.chartPreviousClose, shortName: meta.shortName || `${ticker} Stock`, marketState: meta.marketState || "REGULAR" }, timestamp: data.chart.result[0].timestamp || [Math.floor(Date.now() / 1000)], indicators: data.chart.result[0].indicators } ], error: null } }; return res.json(fakeYahooStructure); } else { return res.status(404).json({ chart: { result: null, error: "Ticker not found" } }); } } catch (e) { console.error("Fetch Error:", e.message); return res.status(500).json({ chart: { result: null, error: e.message } }); } }); const PORT = process.env.PORT || 10000; app.listen(PORT, () => console.log(`Server running on port ${PORT}`));
